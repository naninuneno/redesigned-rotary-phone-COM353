3.3. Explanation of phone formatting code

function validNo() {
  /* if phone has no value, alert for validation, focus on field and return false (not a valid no.) */
  if (!document.userSurvey.phone.value) {
    window.alert("Phone number missing. Please enter a valid phone number to continue.");
    document.userSurvey.phone.focus();
    return false;
  } else {
    var numbersOnly = "";
    var chars = "";
    var phoneNo = document.userSurvey.phone.value;
    /* go through each character in number to evaluate it */
    for (i = 0; i < phoneNo.length; i++) {
      /* get next character in number */
      chars = phoneNo.substring(i, i+1);

      /* if character is valid number (between 1 & 9), add to numbersOnly string */
      if (chars >= "0" && chars <= "9") {
        numbersOnly = numbersOnly + chars;
      }
    }
    /* if there are fewer/more than 13, alert for validation and focus on field, return false */
    if (numbersOnly.length != 13) {
      window.alert("Incorrect phone number format.You must enter 13 numbers.");
      document.userSurvey.focus();
      return false;
    } else {
      /* for valid number, break up into the correct segments */
      var areacode = numbersOnly.substring(0,2);
      var leading0 = numbersOnly.substring(2,3);
      var exchange = numbersOnly.substring(3,5);
      var ext1 = numbersOnly.substring(5,9);
      var ext2 = numbersOnly.substring(9);
      /* use sections of phone number to create new format */
      var newNumber =( "+" + areacode + " " +"(" + leading0 + ")" + exchange + " " +
      ext1 + "-" + ext2);
      /* replace number with correctly formatted number and return true (valid no.) */
      document.userSurvey.phone.value = newNumber;
      return true;
    }
  }
}

3.5. Explanation of email validation code

function validEmail() {
  /* if email has no value, alert with validation message and focus on field */
  if (!document.userSurvey.eMail.value) {
    window.alert("E-mail Address missing. Please enter a valid E-mail address to continue.");
    document.userSurvey.eMail.focus();
    return false;
  } else {
    var emailAddress = document.userSurvey.eMail.value;
    var atLoc = emailAddress.indexOf("@", 1);
    var dotLoc = emailAddress.indexOf(".", atLoc+2);
    var len = emailAddress.length;
    /* validate that there is an '@' symbol in the email address, isn't at position 0 (there is some text before it), that there is a '.', and that there is text at least 2 long after the '.' - typical email address format */
    if (atLoc > 0 && dotLoc > 0 && len > dotLoc+2) {
      return true;
    /* otherwise alert for invalid email address and focus on field */
    } else {
      alert("Invalid E-mail address! Please enter your e-mail address again.");
      document.userSurvey.eMail.focus();
      return false;
    }
  }
}
